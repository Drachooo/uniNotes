    \chapter{Sistemi e segnali}
    \definizione{segnale}{
      Un \textbf{\textcolor{azzurrino}{segnale}} è una porzione di informazione di dati e rappresenta ogni tipo di variabile fisica (scalare o vettoriale) soggetta a variazione.
    }

    Un segnale vive in uno spazio N-dimensionale:
    \begin{itemize}[label=$\triangleright$]
      \item \textbf{immagini}: segnali 2D
      \smallskip
      \item \textbf{video}: segnali 2D + tempo
      \smallskip
      \item \textbf{dati volumetrici (MRI)}: segnali 3D
      \smallskip
      \item \textbf{functional MRI}: seganli 3D + tempo
      \smallskip
      \item \textbf{diffusion MRI}: segnali multidimensionali
    \end{itemize}

    \definizione{sistema}{
      Un \textbf{\textcolor{azzurrino}{sistema}} prende in input un segnale e lo processa per:
      \begin{itemize}[label=$\diamond$]
        \item estrarre informazioni
        \item abilitare trasmissioni tra canali a capacità limitata
        \item migliorare la sicurezza delle reti (crittografia)
        \item formulare diagnosi mediche e pianificare il trattamento (ambito medico)
      \end{itemize}
    }

    La funzione che collega input e output si chiama:
    \begin{itemize}[label=$\Rightarrow$]
      \item \textbf{risposta impulsiva h(t)}: dominio del tempo
      \smallskip
      \item \textbf{funzione di trasferimento H($\boldsymbol{\omega}$)}: dominio delle frequenze
    \end{itemize}

    \section{Tipi di segnali}

    \vario{Segnale a tempo continuo VS segnale a tempo discreto}{
      Un \textbf{\textcolor{azzurrino}{segnale a tempo continuo}}, se esiste, per ogni valore reale della variabile indipendente.

      Il dominio della funzione che rappresenta il segnale avrà la stessa cardinalità dell'insieme dei numeri reali $\mathbb{R}$.

      \medskip
      Un \textbf{\textcolor{azzurrino}{segnale a tempo discreto}}, se esiste,  solo per alcuni valori della variabile indipendente con cardinalità pari a quella dell'insieme dei numeri interi $\mathbb{Z}$.
    }

    \vario{Segnale analogico VS segnale digitale}{
      Un \textbf{\textcolor{azzurrino}{segnale}} è \textbf{\textcolor{azzurrino}{analogico}} se la sua ampiezza comprende tutti i valori in un range continuo e il codominio della sua funzione avrà la stessa cardinalità dell'insieme dei numeri reali $\mathbb{R}$.

      \medskip
      Un \textbf{\textcolor{azzurrino}{segnale}} è \textbf{\textcolor{azzurrino}{digitale}} se la sua ampiezza comprende un numero finito di valori n-ario.
    }

    Un segnale analogico equivale a un segnale a tempo continuo, ma sull'asse y.

    Un segnale digitale corrisponde a un segnale a tempo discreto, ma sull'asse y.

    \vario{Segnale periodico VS segnale a-periodico}{
      Un \textbf{\textcolor{azzurrino}{segnale}} f(t) è \textbf{\textcolor{azzurrino}{periodico}} se esiste una costante positiva $T_0$ tale che:
      \begin{equation*}
        f(t+T_0) = f(t) \ \ \forall t
      \end{equation*}

      Il valore minore di $T_0$ è detto \hl{periodo} di f(t).
    }

    \vario{Segnale casuale VS segnale anti-casuale VS segnale non-causale}{
      Un \textbf{\textcolor{azzurrino}{segnale}} è \textbf{\textcolor{azzurrino}{casuale}} se ha valore nullo per ogni tempo negativo:
      \begin{equation*}
        f(t) = 0 \ \ \forall t < 0
      \end{equation*}

      Un \textbf{\textcolor{azzurrino}{segnale}} è \textbf{\textcolor{azzurrino}{anti-casuale}} se ha valore nullo per ogni tempo positivo:
      \begin{equation*}
        f(t) = 0 \ \ \forall t \geq 0
      \end{equation*}

      Un \textbf{\textcolor{azzurrino}{segnale}} è \textbf{\textcolor{azzurrino}{non-casuale}} se non ha mai valore nullo, sia per tempi negativi che positivi:
      \begin{equation*}
        f(t) \neq 0 \ \ \forall t
      \end{equation*}
    }

    \vario{Segnale pari VS segnale dispari}{
      Un \textbf{\textcolor{azzurrino}{segnale}} f(t) è \textbf{\textcolor{azzurrino}{pari}} se vale:
      \begin{equation*}
        f(t) = f(-t) \ \ (es. coseno)
      \end{equation*}

      Un \textbf{\textcolor{azzurrino}{segnale}} f(t) è \textbf{\textcolor{azzurrino}{dispari}} se vale:
      \begin{equation*}
        f(t) = -f(-t) \ \ (es. seno)
      \end{equation*}
    }

    Ogni segnale può essere riscritto come combinazione di un segnale pari e un segnale dispari.

    \vario{Segnale determinisco VS segnale random}{
      Un \textbf{\textcolor{azzurrino}{segnale}} è \textbf{\textcolor{azzurrino}{determinisco}} se può essere descritto e previsto in ogni istante di tempo matematicamente o graficamente (es. seno, coseno).

      \medskip
      Un \textbf{\textcolor{azzurrino}{segnale}} è \textbf{\textcolor{azzurrino}{random}} se non può essere descritto in ogni istante di tempo e sono caratterizzati da variabilità.

      Ogni segnale che deriva da una misurazione è un segnale random (ECG, audio, immagini scanerizzate).
    }

    \vario{Segnale finito VS segnale infinito}{
      Un \textbf{\textcolor{azzurrino}{segnale}} è \textbf{\textcolor{azzurrino}{finito}} se possiede un dominio finito:
      \begin{equation*}
        f = f(t), \forall t : t_1 < t < t_2
      \end{equation*}

      Un \textbf{\textcolor{azzurrino}{segnale}} è \textbf{\textcolor{azzurrino}{infinito}} se possiede un dominio infinito:
      \begin{equation*}
        f(t) = sin(\omega t)
      \end{equation*}
    }
