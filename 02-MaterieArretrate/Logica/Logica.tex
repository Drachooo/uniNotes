  \chapter*{Introduzione alla logica}
  \addcontentsline{toc}{chapter}{Introduzione alla logica}
  La \textbf{logica} si occupa di \hl{formalizzare i ragionamenti matematici} utilizzando il linguaggio matematico.

  Ciò è dovuto al fatto che il linguaggio naturale è ambiguo per la descrizione di teorimi.

  La logica si divide in:
  \begin{itemize}
    \item \textbf{logica formale}: che a sua volta si divide in
    \begin{itemize}
      \item \textit{logica simbolica}
      \item \textit{Logica matematica}: studia la calcolabilità [studiata a fondamenti] 
    \end{itemize}
    \item \textbf{logica filosofica}: ha a che fare con il linguaggio naturale
  \end{itemize}

  \section*{Numeri naturali}
  \addcontentsline{toc}{section}{Numeri naturali}
  Dal punto di vista intuitivo i numeri sono un insieme infinito:
  \begin{equation*}
    \mathbb{N} = \{0, 1, 2, \dots\}
  \end{equation*}
  per i quali sono definite le operazioni standard aritmetiche elementari.

  Peano definì in maniera rigorosa i numeri naturali.

  \definizione{numeri naturali}{I \textbf{numeri naturali} sono una tupla:
    \begin{equation*}
      \langle \mathbb{N}, 0, succ \rangle
    \end{equation*}

    Su quetsa tupla, sono definite 4 assiomi (teoremi che non si dimostrano):
    \begin{enumerate}
      \item $0 \in \mathbb{N}$
      \item $succ: \mathbb{N} \rightarrow \mathbb{N}$ (operazione unaria iniettiva)
      \item $0 \notin \mathrm{Im}(succ)$ (zero non è successore di nessuno)
      \item (\textbf{Principio di induzione}) per ogni $\mathbb{P} \subseteq \mathbb{N}$ valgono le seguenti proprietà:
      \begin{itemize}
        \item $0 \in \mathbb{P}$
        \item $\forall n \in \mathbb{N}, (n \in \mathbb{P} \rightarrow succ(n) \in \mathbb{P})$
      \end{itemize}
      $\Rightarrow \mathbb{P} = \mathbb{N}$
    \end{enumerate}}


  \chapter*{Logica proposizionale}
  \addcontentsline{toc}{chapter}{Logica proposizionale}
  
  La logica è lo studio del ragionamento deduttivo su delle entità linguistiche che indentifichiamo essere le proposizioni.\\
  La logica classica è la branca che studia le poposizioni: frasi che possono avere due valori di verità, \textbf{VERO} o \textbf{FALSO}.


  \section*{Sintassi}
  \addcontentsline{toc}{section}{Sintassi}

  Esempi di proposizione:
  \begin{enumerate}
    \item Verona è una città del Veneto $\rightarrow$ \textbf{\textcolor{green!50!black}{proposizione VERA}}
    \item Glu stidenti di questa classe sono al primo anno $\rightarrow$ \textbf{\textcolor{red}{proposizione FALSA}}
    \item Fuori sta piovendo $\rightarrow$ \textbf{\textcolor{green!50!black}{proposizione FALSA}}
    \item Come ti chiami? $\rightarrow$ \textbf{\textcolor{orange}{NON è una proposizione}}
  \end{enumerate}

  Vogliamo rendere questo linguaggio formale $\rightarrow$ bisogna individuare dei simboli.\\
  Una volta identificate le assunzioni, dobbiamo rappresentarle mediante simboli\dots \\es: P, Q, R

  \textbf{Esempio}:\\
  \uline{Se c'è il sole}, \uline{allora devi mettere la crema}

  \definizione{Alfabeto $\mathcal{L}$}{L'alfabeto $\mathcal{L}$ della logica prop ha i seguenti simboli:
  \begin{itemize}
    \item insieme numberabile di simboli proposizionali: \textbf{$AT$}$_{prop}$ = \{$p_0$, $p_1$, \dots\}
    \item connettivi: $\land$, $\lor$, $\rightarrow$, $\bot$
    \item simboli ausiliari: servono per costruire il linguaggio ma non ne fanno parte, per esempio le parentesi tonde ()
  \end{itemize}
  }
  Le proposizioni semplici vengono definite \textbf{atomi} oppure \textbf{proposizioni atomiche}.

  \definizione{Appartenenza insieme prop}{L'insieme prop delle proposizioni è definito come il più piccolo insieme P di stringhe su $\mathcal{L}$ tali che:
  \begin{itemize}
    \item $AT_{prop}$ $\subseteq$ P.
    \item $\bot \in$ P.
    \item se $\phi, \psi \in$ P allora $(\phi \land \psi), (\phi \lor \psi), (\phi \rightarrow \psi) \in P$
  \end{itemize}}

  \textbf{Esempio:}\\
  $(p_1 \rightarrow p_2)$ $\in$ PROP \qquad\qquad\qquad \textcolor{green!50!black}{$\rightarrow$ Completa}\\
  $p_7 \lor p_4 \notin$ PROP \qquad\qquad\qquad\qquad \textcolor{red}{$\rightarrow$ Mancano le parentesi}\\
  $((\bot \lor p_7) \land (\lnot p_4)) \in$ PROP \qquad\qquad \textcolor{green!50!black}{$\rightarrow$ Completa}\\

  \notazione{Chiamiamo AT = $AT_{prop} \cup \{\bot\}$
  \begin{enumerate}[label= \alph*)]
    \item $\lnot \phi$ denotiamo $(\phi \rightarrow \bot)$
    \item $\phi \equiv \psi$ denotiamo $((\phi \rightarrow \psi) \land (\psi \rightarrow \phi))$
  \end{enumerate}
  }

  \convenzione{
  \begin{enumerate}
    \item Omettiamo le parentesi più esterne \qquad $(\psi \rightarrow \phi)$ diventa $\psi \rightarrow \phi$
    \item Scriviamo $\lnot \psi$ anzichè $(\lnot \psi)$
    \item $\land$ e $\lor$ legano più forte di $\rightarrow$ e $\equiv$ \quad $\phi_1 \land \phi_2 \rightarrow \psi_1 \lor \psi_2$ sta per $(\phi_1 \land \phi_2) \rightarrow (\psi_1 \lor \psi_2)$
  \end{enumerate}
  }


  \subsection*{Funzioni ricorsive su Prop}
  \addcontentsline{toc}{subsection}{Funzioni ricorsive}
  \definizione{Funzione $\mathscr{L}$ lunghezza}{Funzione $\mathscr{L}$ :PROP $\rightarrow \mathbb{N}$ che ad ogni $\phi \in PROP$ associa la sua lunghezza
  \vspace{0,2cm}
  \begin{enumerate}
    \item $\mathscr{L}$($\phi$) = 1 \qquad se $\phi \in AT$
    \item $\mathscr{L}$($\phi$ o $\psi$) = 3 + $\mathscr{L}(\phi)$ + $\mathscr{L}(\psi)$ \qquad con o $\in$ $\{\land, \lor, \rightarrow\}$
    \item $\mathscr{L}$($\lnot \phi$) = 3 + $\mathscr{L}(\phi)$
  \end{enumerate}
  il \textbf{+3} indica le due parentesi tonde e il connettivo
  }  

  \esercizio[Definiamo il numero di parentesi tonde in una formula]{b($\varphi$) $\rightarrow$ numero di parentesi in una formula proposizionale
  \begin{itemize}
    \item \textcolor{red}{b(}$\varphi$\textcolor{red}{)} = 0 \qquad  \qquad \qquad con $\varphi \in$ AT
    \item \textcolor{red}{b(}($\varphi$ o $\phi$)\textcolor{red}{)} = 2 + b($\varphi$) \qquad con o $\in$ \{$\land, \lor, \rightarrow$ \}
    \item \textcolor{red}{b(}($\lnot \varphi$)\textcolor{red}{)} = 2 + b($\varphi$) 
  \end{itemize}}

  \esercizio[Definizione di parsing tree]{Definiamo il parsing tree di una proposizione
  \begin{itemize}
    \item T($\varphi$) =  $\cdot \varphi$     \qquad con $\varphi$ $\in$ AT
    \item T($\varphi$ o $\psi$) = $\cdot$(T($\varphi$), T($\psi$))    \qquad \qquad con o $\in$ \{ $\land$, $\lor$, $\rightarrow$\}
    \item T($\lnot \varphi$) = $\lnot$(T($\varphi$))
  \end{itemize}
  }
  

  Una volta definito costruiamo il \textit{parsing tree}\footnote{Parsing tree: visualizzazione ad albero di una stringa} della proposizione\\
  \begin{forest}
    [$(p_1 \rightarrow (\bot \lor (\lnot p_3)))$
      [$p_1$]
      [$\bot \lor (\lnot p_3)$
        [$\bot$]
        [$\lnot p_3$
        [$p_3$]
        ]
      ]
    ]
  \end{forest}


  \definizione{Rango $r$ di una formula}{
    \begin{itemize}
      \item $r$($\varphi$) = 0  \qquad  \qquad $\varphi \in$ AT
      \item $r$($\varphi$ o $\psi$) = 1 + max\{$r$($\varphi$), $r$($\psi$)\} \qquad con o $\in$ \{$\lor$, $\land$, $\rightarrow$\}
      \item $r$($\lnot \varphi$) = 1 + r($\varphi$)
    \end{itemize}
  }

  \definizione{Sottoformula Sub}{
    \begin{itemize}
      \item Sub($\varphi$) = \{$\varphi$\}   \qquad $\varphi \in$ AT
      \item Sub($\varphi$ o $\psi$) = Sub($\varphi$) $\cup$ Sub($\psi$) $\cup$ \{$\varphi$ o $\psi$\} \qquad con o $\in$ $\{\land, \lor, \rightarrow\}$
      \item Sub($\lnot \varphi$) = Sub($\varphi$) $\cup$ \{($\lnot \varphi$)\} 
    \end{itemize}
  }


  \esercizio[Calcola il rango della seguente espressione:\\ ($\lnot p_2 \rightarrow$($p_3$ $\lor$ ($p_1 \leftrightarrows p_2$)))$\land$ $\lnot p_3$]{
  $r$($\varphi \land \psi$) = 1 + max\{($\lnot p_2 \rightarrow$ ($p_3 \lor$($p_1 \leftrightarrow p_2$))), $\lnot p_3$\}\\
  $r$($\varphi \land \psi$) = 1 + max\{1+max\{1+0, $p_3 \lor$($p_1 \leftrightarrow p_2$)\}, 1+0\}\\
  $r$($\varphi \land \psi$) = 1 + max\{1+max\{1,1+max\{0,0\}\},1\}\\
  $r$($\varphi \land \psi$) = 1 + max\{1+max\{1,1\},1\}\\
  $r$($\varphi \land \psi$) = 1 + max\{2,1\} = 1+2 = 3}
  

  \definizione{Numero di connettivi \#($\varphi)$}{
    \begin{itemize}
      \item \#($\varphi)$ = 0
      \item \#($\varphi \lor \psi$) = 1 + \#$\varphi$ + \#$\psi$
      \item \#($\lnot \varphi)$ = 1 + \#$\varphi$
    \end{itemize}
  }

  \section*{Semantica}
  \addcontentsline{toc}{section}{Semantica}
  La \textbf{semantica} si occupa di \hl{attribuire un significato sintattico} alle proposizioni (vero - 1 o falso - 0).


  \definizione{valutazione di verità}{E' una funzione $\sigma: PROP \rightarrow \{{0, 1}\}$ tale che:
    \begin{itemize}
      \item $\sigma(\bot) = 0$
      \item $\sigma(\lnot \varphi) = 1$ se e solo se $\sigma(\varphi) = 0$
      \item $\sigma(\phi \land \psi) \Leftrightarrow \sigma(\phi) = \sigma(\psi) = 1$
      \item $\sigma(\phi \lor \psi) \Leftrightarrow \sigma(\phi) = 1$ oppure $\sigma(\psi) = 1$
      \item $\sigma(\phi \rightarrow \psi) \Leftrightarrow \sigma(\phi) = 1$ allora $\sigma(\psi) = 1$
    \end{itemize}
    }
  

  Questa definizione per la funzione di valutazione è pressochè la formulzazione matematica delle \textbf{tavole di verità}.

  \definizione{valutazione atomica}{Funzione $v: AT \rightarrow \{0, 1\}$ tali che $v(\bot) = 0$ 
  }

  \proposizione{Per ogni valutazione $v$ esiste ed è unica una \textit{valutazione di verità} ${\llbracket - \rrbracket}_v$ tale che per ogni $p \in \mathbf{AT}$, ${\llbracket p \rrbracket}_v = v(p)$.
  }

  \definizione{concordanza}{  Dato un insieme finito $P$ di simboli proposizionali, si dice che due valutazioni $v, v'$ \textbf{concordano} su $P$ se $v_{|P} = v'_{|P}$. 
  }

  La proposizione precedente ci dice che l'insieme di tutte le possibili valutazione ha le stessa cardinalità di $\mathbb{N}$.

  Un'altra proprietà delle valutazione è stabilita dalla seguente proposizione.

  \proposizione{Se $P(\phi)$ è l'insieme dei simboli proposizionali che occorrono in una formula $\phi$ e $v, v'$ due valutazioni che concordano su $P(\phi)$ allora ${\llbracket \phi \rrbracket}_v = {\llbracket \phi \rrbracket}_{v'}$.
  }

  Se un problema è vero in tutte le sue valutazioni, allora è \textbf{decidibile} e viene detto \textbf{tautologia}.

  Si estende la valutazione di verità a un insieme arbitrario di proposizioni $\Gamma$: \par
  \begin{center}
      ${\llbracket \Gamma \rrbracket}_v = 1$ se e solo se per ogni $\alpha \in \Gamma$, ${\llbracket \alpha \rrbracket}_v = 1$
  \end{center}

  Tuttavia si pone attenzione al fatto che se ${\llbracket \phi \rrbracket}_v \neq 1$, allora non significa che per ogni $\alpha \in \Gamma$ si ha ${\llbracket \alpha \rrbracket}_v = 0$, al contrario si intende che esiste almeno un $\alpha \in \Gamma$ tale per cui ${\llbracket \alpha \rrbracket}_v = 0$.

  Viceversa è più forte affermare che ${\llbracket \Gamma \rrbracket}_v = 0$, in quanto per questo caso ogni $\alpha \in \Gamma$ ha valutazione ${\llbracket \alpha \rrbracket}_v = 0$.

  Partendo dal concetto di valutazione dell formule, è possibile definire cosa significhi che da un insieme $\Gamma$ di proposizioni \textbf{segue logicamente} una proposizione $\varphi$, o in altre parole che $\varphi$ sia \textbf{conseguenza logica} di $\Gamma$.

  \definizione{conseguenza logica}{Dato un insieme $\Gamma$ di formule, si dice che la formual $\varphi$ è conseguenza logica dalle formule in $\Gamma$ e si scrive $\Gamma \models \varphi$, se per ogni valutazione $v$, ${\llbracket \Gamma \rrbracket}_v = 1 \Rightarrow {\llbracket \varphi \rrbracket}_v = 1$.
  }

  La formula $\varphi$ è detta \textbf{soddisfacibile} se esiste una valutazione $v$ tale che ${\llbracket \varphi \rrbracket}_v = 1$, altrimenti risulta essere \textbf{insoddisfacibile}.

  A sua volta un insieme di formule $\Gamma$ è soddisfacibile se esiste una valutazione $v$ tale che per ogni $\varphi \in \Gamma$ si ha ${\llbracket \Gamma \rrbracket}_v = 1$, altrimenti si dice insoddisfacibile.

  \notazione{\begin{itemize}
        \item si scrive $\models \varphi$ al posto di $\phi \models \varphi$
        \item si scrive $\psi_1, \dots, \psi_n \models \varphi$ al posto di $\{\psi_1, \dots, \psi_n\} \models \varphi$
        \item $\Gamma, \Sigma, \Delta$ sono insiemi arbitrari di formule
        \item si scrive $\Gamma, \Sigma, \psi \models \varphi$ al posto di $\Gamma \cup \Delta \cup \{\psi\} \models \varphi$ 
    \end{itemize}
  }


  \esercizio[$\models \lnot\lnot\varphi \rightarrow \varphi$ è una tautologia? ]{
    Dimostrazione: per ogni valutazione v.${\llbracket \lnot\lnot\varphi \rightarrow \varphi \rrbracket}_v$ = 1\par
    ${\llbracket \lnot\lnot\varphi \rightarrow \varphi \rrbracket}_v$ = 1 \qquad $\leftrightarrow$ \sottolinea{green}{$\llbracket\lnot\lnot\varphi\rrbracket_v = 0$} oppure $\llbracket\varphi\rrbracket_v = 1$\\

  \subsection*{Proprietà di $\models$}
\addcontentsline{toc}{subsection}{\texorpdfstring{Proprietà di $\models$}{Proprietà di ⊨}}

  \begin{enumerate}
    \item $\Gamma \models \psi$ e $\bigtriangleup.\psi \models \varphi \Rightarrow  \Gamma, \bigtriangleup \models \varphi$
    \item $\Gamma \models \varphi \lor \psi$ e $\bigtriangleup, \varphi \models \delta$ e $\varepsilon, \psi \models \delta \Rightarrow \Gamma, \bigtriangleup, \varepsilon \models \Gamma$ 
  \end{enumerate}

  \textbf{\large Dimostrazione 1}
  \begin{itemize}[label=-]
    \item $\Gamma \models \psi \leftrightarrow$ Per ogni valutazione v.$\llbracket \Gamma \rrbracket_v = 1 \rightarrow \llbracket \psi \rrbracket_v = 1$
    \item $\bigtriangleup, \psi \models$ $\leftrightarrow$ Per ogni valutazione v.$\llbracket \bigtriangleup \rrbracket_v$ = 1 e $\llbracket \psi \rrbracket_v$ = 1 $\rightarrow \llbracket \varphi \rrbracket_v$ = 1\\
    $\Rightarrow \llbracket \Gamma, \bigtriangleup \rrbracket_v$ = 1 $\Rightarrow \llbracket \varphi \rrbracket_v$ = 1 
  \end{itemize}

  \textbf{\large Dimostrazione 2}\par
  \textcolor{orange}{Ipotesi:} \sottolinea{orange}{$\llbracket \Gamma \rrbracket_v = 1 = \llbracket \bigtriangleup \rrbracket_v = \llbracket \varepsilon \rrbracket_v \Rightarrow \llbracket \delta \rrbracket_v = 1$}\par
  \hspace*{1,40cm} $\llbracket \varphi \lor \psi \rrbracket_v = 1$\par
  \hspace*{1,40cm} $\llbracket \varphi \rrbracket_v = 1$ OR $\llbracket \psi \rrbracket_v = 1$\par

  Bisogna procedere per casi:
  \begin{itemize}
    \item Se $\llbracket \varphi \rrbracket_v = 1$ allora siccome $\llbracket \bigtriangleup \rrbracket_v = 1$ allora $\llbracket \delta \rrbracket_v = 1$
    \item Se $\llbracket \psi \rrbracket_v = 1$ allora siccome $\llbracket \varepsilon \rrbracket_v = 1$ allora $\llbracket \delta  \rrbracket_v = 1$
  \end{itemize}

  \vspace{0,5cm}

  \esercizio[Utilizziamo le regole semantiche per dimostrare:\\
  \sottolinea{gray}{$\varphi_1 \land \varphi_2 \leftrightarrow \varphi_1' \land \varphi_2'$} che è come scrivere \sottolinea{gray}{$\varphi_1 \land \varphi_1' \leftrightarrow \varphi_2 \land \varphi_2'$}]{
    \begin{itemize}
      \item $\models \varphi_1 \land \varphi_2 \rightarrow \varphi_1' \land \varphi_2'$ \quad ovvero \quad $\llbracket \varphi_1 \land \varphi_2 \rrbracket_v = 1 \Rightarrow \llbracket \varphi_1' \land \varphi_2'\rrbracket_v = 1$
    \end{itemize}
    \textbf{\large Dimostrazione:} \quad $\llbracket \varphi_1 \rrbracket_v = 1$ e $\llbracket \varphi_2 \rrbracket_v = 1$ \quad $\Rightarrow \quad \llbracket \varphi_1' \land \varphi_2' \rrbracket_v = 1$
    \begin{itemize}
      \item     $\llbracket \varphi_1\rrbracket_v = 1$ e \sottolinea{green}{$\llbracket \varphi_1 \land \varphi_1' \rrbracket_v =1$} $\quad \Rightarrow \quad \llbracket \varphi_1' \rrbracket_v = 1$\\
      \sottolinea{green}{$\llbracket \varphi \rrbracket_v = 1$} $\Rightarrow \llbracket \varphi_1' \rrbracket_v = 1$
      \item     $\llbracket \varphi_2 \rrbracket_v = 1$ e \sottolinea{red}{$\llbracket \varphi_2 \rightarrow \varphi_2' \rrbracket_v = 1$} $\Rightarrow \llbracket \varphi_2' \rrbracket_v = 1$\par
      \sottolinea{red}{$\llbracket \varphi_2 \rrbracket_v = 1$} $\Rightarrow \llbracket \varphi_2' \rrbracket_v = 1$\par
      $\Rightarrow \llbracket \varphi_1' \land \varphi_1' \rrbracket$
    \end{itemize}
  }

  \definizione{Sostituzione}{Siano $\psi$ e $\gamma$ due formule e sia p $\in$ Prop. Si definisce $\psi[\textbf{p}/\gamma]$ la formula ottenuta sostituendo nella stringa $\psi$ ogni occorrenza di \textbf{p} con la stringa corrispondente a $\psi$.}

  \teorema[Sostituzione]{Siano $\varphi, \gamma_1. \gamma_2 \in$ Prop e sia p una proposizione atomica.\\
  Se $\models \gamma_1 \leftrightarrow \gamma_2$ allora $\models \varphi[p/\gamma_1] \leftrightarrow \varphi[p/\gamma_2]$}{(per induzione) su $\varphi$\\
  \sottolinea{black}{Base: Sia $\varphi \in \textbf{AT}$. Abbiamo due casi:}
  \begin{enumerate}
    \item se $\varphi \neq \textbf{p} $ allora $\varphi [\textbf{p}/\gamma_1] = \varphi[\textbf{p}/\gamma_2] = \varphi$ e quindi ovviamente $\models \varphi \equiv \varphi$
    \item se $\varphi = \textbf{p}$ allora $\varphi[\textbf{p}/\gamma_1] = \gamma_1$ e $\varphi[\textbf{p}/\gamma_2] = \gamma_2$ e quindi per ipotesi $\models \varphi[\textbf{p}/\gamma_1] \equiv \varphi[\textbf{p}/\gamma_1]$
  \end{enumerate}
  
  \vspace{2em}
  
  \sottolinea{black}{Passo induttivo: Esaminiamo i vari casi}
  \begin{description}
    \item[($\varphi = \varphi_1 \rightarrow \varphi_2$)] Per ipotesi induttiva:\\ $\models \varphi_1[\textbf{p}/\gamma_1] \equiv \varphi_1[\textbf{p}/\gamma_2]$ e $\models \varphi_2[\textbf{p}/\gamma_1] \equiv \varphi_2[\textbf{p}/\gamma_2].$\\
    Ora abbiamo che $\models \varphi_1[\textbf{p}/\gamma_1] \rightarrow \varphi[\textbf{p}/\gamma_1] \equiv \varphi_1[\textbf{p}/\gamma_2] \rightarrow \varphi_2[\textbf{p}/\gamma_2].$\\ 
    Dato che $\varphi[\textbf{p}/\gamma_1] = \varphi_1[\textbf{p}/\gamma_1] \rightarrow \varphi_2[\textbf{p}/\gamma_1]$ e $\varphi[\textbf{p}/\gamma_2] = \varphi_1[\textbf{p}/\gamma_2] \rightarrow \varphi_2[\textbf{p}/\gamma_2]$
    \item[($\varphi = \varphi_1 \land \varphi_2$), ($\varphi = \varphi_1 \lor \varphi_2$)] Il ragionamento è analogo al caso precedente
  \end{description}
  }

  Come risolvere le tautologie?
  \begin{itemize}
    \item Metodo 1: Utilizzare le tavole di verità $\rightarrow$ dispendioso
    \item Metodo 2: Utilizzare la semantica $\rightarrow$ da utilizzare in esame
  \end{itemize}

  $\models \delta \Leftrightarrow \forall$v $\llbracket \delta \rrbracket_v =1 \Leftrightarrow (\llbracket \varphi \rrbracket_v = 1 \rightarrow$ \sottolinea{green}{$\llbracket (\psi \rightarrow \sigma) \rightarrow$ \textbf{(} $(\varphi \rightarrow \psi) \rightarrow (\varphi \rightarrow \sigma)$ \textbf{)} $\rrbracket_v = 1$})\\

  \section*{Sistemi deduttvi}
  \addcontentsline{toc}{section}{Sistemi deduttivi}
  L'approcio non semantico alla logica proposizionale si basa sul concetto di \textbf{dimostrazione formale} come \textit{deduzione}.

  \definizione{sistema deduttivo}{Fornisce un insieme di regole per costruire dimostrazioni formali che permettono di derivare una conclusione a partire da alcune premesse.}

  Quando si enuncia un \textbf{teorema} deve sempre esistere la sua \textbf{dimostrazione}.

  \definizione{dimostrazione}{Sequenza finita di passi logici formalemte corretti che permettono di arrivare alla \textbf{tesi} a partire dalle \textbf{ipotesi} (note anche come \textbf{premesse} o \textbf{assunzioni}).}

  \[
    \left.
    \begin{array}{c}
    \text{Ipotesi / premesse / assunzioni} \\
    \vdots \\
    \text{Tesi}
    \end{array}
    \right\}
    \text{Dimostrazione / Derivazione}
  \]

  Tra le ipotesi e la tesi esiste una serie finita di passi indicati con \vdots che consete la cosidetta \textbf{deduzione}.

  \subsection*{Deduzione naturale}
  \addcontentsline{toc}{subsection}{Deduzione naturale}
  La \textbf{deduzione naturale} è una metologia di ragionamento per costruire dimostrazioni formali basata su regole di inferenza che riflettono il modo naturale di ragionare.

  Per tanto non si valutano più i valori di verità delle formule, ma si considera la loro struttura sintattica. stabilendeo delle:
  \begin{itemize}
    \item \textbf{assunzioni}: prese temporaneamente come vere
    \item \textbf{derivazioni} sulle formule applicando delle leggi: l'applicazione di una regola si chiama \textit{passo di derivazione}
    \item \textbf{scaricamento} delle assunzioni: quando si arriva alla tesi, le assunzioni vengono eliminate
  \end{itemize}

  Tutte le derivazioni vengono rappresentate graficamente tramite gli \textbf{alberi di derivazione}, nei quali:
  \begin{itemize}
    \item \textbf{foglie}: rappresentano le \textit{ipotesi}
    \item \textbf{nodi interni}: rappresentano i \textit{passi di derivazione}
    \item \textbf{radice}: rappresenta la \textit{tesi} da dimostrare
  \end{itemize}

  \definizione{derivazione completa}{Una derivazione si dice completa se non ci sono più foglie aperte (tutte le assunzioni sono state scaricate).}

  \esempio{Dimostrare $p \land q \vdash q \land p$}{
    \begin{forest}
      [$q \land p$
        [$q$
          [$p \land q$, edge label={node[midway,left]{Assunzione}}]
          , edge label={node[midway,left]}]
        [$p$
          [$p \land q$, edge label={node[midway,right]{Assunzione}}]
          , edge label={node[midway,right]}]
        , edge label={node[midway,left]}]
    \end{forest}
  }

  \subsection*{Regole deduttive per i connettivi}
  \addcontentsline{toc}{subsection}{Regole deduttive per i connettivi}
  Le \textbf{regole di derivazione} si divisono in due tipologie:
  \begin{itemize}
    \item \textbf{regole di introduzione}: le premesse vengono collegate da un connettivo logico introdotto
    \item \textbf{regole di eliminazione}: dalle premesse si può eliminare un connettivo logico
  \end{itemize}

  \[
    \begin{array}{c}
    \text{premesse} \\
    \hline
    \text{conclusione}
    \end{array}%
    \quad \text{(nome regola)}
  \]

  Le formule presenti all'interno delle parentesi [ ] sono delle ipotesi scaricate (assunte) da una specifica regola di derivazione utilizzata.
  
  Per specificare a quale regola si riferiscono è necessario indicare un indice numerico.

\subsection*{Correttezza e completezza}
\addcontentsline{toc}{subsection}{Correttezza e completezza}


\subsubsection*{Correttezza}
\addcontentsline{toc}{subsubsection}{Correttezza}
Il primo passo da eseguire è quello di dimostrare che la deduzione naturale sia corretta.

La \textbf{correttezza} consiste che se da $\Gamma$ si deriva $\varphi \ (\Gamma \vdash \varphi)$, allora $\varphi$ è conseguenza logica logica di $\Gamma \ (\Gamma \models \varphi)$.

\lemma{Correttezza}{Per ogni $\varphi$ e per tutte derivazioni in $\Pi$, se $\underset{\varphi}{\Pi}$ allora Hp($\Pi$) $\models \varphi$.}{}

La dimostrazione procede per induzione sull'altezza delle derivazioni: \\[0.2ex]
$\Rightarrow$ \textbf{caso base}: se $\Pi$ = $\varphi$, allora dato che $\varphi \models \varphi$ abbiamo la tesi \\[0.2ex]
$\Rightarrow$ \textbf{passo induttivo}: per casi sull'ultima regola della derivazione $\Pi$

\teorema[Correttezza / Soundness]{\begin{equation*}
\Gamma \vdash \varphi \Rightarrow \Gamma \models \varphi
\end{equation*}}{\\[0.2ex]Se $\Gamma \vdash \varphi$ allora esiste una derivazione $\underset{\varphi}{\Pi}$ tale che Hp[$\Pi$] $\subseteq \Gamma$. \\[0.2ex]
Per il Lemma 2 abbiamo che Hp[$\Pi$] $\models \varphi$ e per il Lemma 1 concludiamo che:
\begin{equation*}
\Gamma \models \varphi
\end{equation*}}

Una conseguenza del teorema di correttezza è la \hl{\textbf{non} contraddittorietà} della deduzione naturale.

\teorema[]{\begin{equation*}
\textcolor{red}{\not} \vdash
\end{equation*}}{\\[0.2ex]Supponiamo che $\vdash \bot$, allora per la correttezza avremmo che $\models \bot$ che è impossibile.}

\subsubsection*{Completezza}
\addcontentsline{toc}{subsubsection}{Completezza}
La \textbf{completezza} \hl{consiste nel dimostare che $\Gamma \models \varphi \Rightarrow \Gamma \vdash \varphi$: se una formula è corretta, allora è anche derivabile}.

Per dimostarlo, dobbiamo utilizzare due concetti:
\begin{enumerate}
  \item consistenza
  \item soddisfacibilità
\end{enumerate}

\definizione{consistenza}{Si dice che un insieme $\Gamma \subseteq PROP$ è \textbf{incosistente} se $\Gamma \vdash \bot$, altrimenti si dice \textbf{consistente} ($\Gamma \nvdash \bot$).}

\hl{\textbf{Soddisfacibilità} e \textbf{consistenza} sono fortemente legati tra di loro}. \\[0.2ex]
L'obiettivo principale è quello di dimostare che ogni insieme conistente di formule è soddisfacibile, ma è facile dimostare l'implicazione inversa (ogni insieme soddisfacibile di formule è consistente).

\Proposizione{\begin{equation*}
\Gamma \vdash \bot \Rightarrow \forall v, \llbracket \Gamma \rrbracket_v \neq 1
\end{equation*}}{\\[0.2ex]Se $\Gamma \vdash \bot$ allora, per la correttezza, $\Gamma \models \bot$ e, quindi, per la definizione di conseguenza logica non esiste nessuna valutazione $v$ tale che $\llbracket \Gamma \rrbracket_v = 1$.}

Per contrapposizione abbiamo, quindi che se $\Gamma$ è soddisfacibile allora $\Gamma$ è consistente. \\

Introduciamo un concetto un pò più forte della consistenza: \textbf{consistenza massimale}.

\definizione{insieme consistente massimale (opzione 1)}{Sia $\mathcal{C} = \{ \Gamma : \Gamma \subseteq PROP$ e $\Gamma \nvdash \bot \}$. \\[0.2ex] Si definiscono \textbf{massimali consistenti} gli elementi massimali dell'ordine parziale $\langle \mathcal{C}, \subseteq \rangle$.}


\definizione{insieme consistente massimale (opzione 2)}{Un insieme di formule $\Gamma$ si dice \textbf{massimale consistente} se è consistente e non esiste un insieme $\Delta \neq \Gamma$ tale che $\Gamma \subseteq \Delta$.}

Una caratteristica importante degli insiemi consistenti di formule è che possono estesi ad insiemi massimali consistenti.

\Proposizione{\textit{Ogni insieme consistente} $\Gamma$ \textit{è contenuto in (almeno) un insieme massimale consistente}.}{\\[0.2ex]Fissiamo un'enumerazione $(\varphi_i)_{i < \omega}$ di tutte le formule proposizionali. \\[0.2ex]

Definiamo induttivamente la seguente sucessione $(\Gamma_i)_{i < \omega}$ di insiemi di formule:
\begin{itemize}
  \item $\Gamma_0 = \Gamma$
  \item $\Gamma_{i+1}$ = $\begin{cases}
    \Gamma_i \cup {\Gamma_i} \cup {\varphi_i} \nvdash \bot \\
    \Gamma_i \ \text{altrimenti}
  \end{cases}$
\end{itemize}

Dimostriamo che:
\begin{enumerate}
  \item $\forall \ i. \ \Gamma_i \subseteq \Gamma_{i+1} \Rightarrow \text{verifico correttezza}$ (vero per costruzione)
  \item $\forall \ i. \ \Gamma_i \nvdash \bot \Rightarrow$ si parte da un insieme consistente e ad ogni passo della costruzione preserva la consistenza (vero per induzione)
  \item $\Gamma$* = $\cup_{i \in \mathbb{N}} \ \Gamma_i$ è massimale conistente
\end{enumerate}

Per quest'ultimo caso, dimostriamo per prima cosa che $\Gamma$* è \textit{consistente}.

Supponiamo che $\Gamma$* $\vdash \bot$, allora esiste un sottoinsieme finito $\Sigma$ di $\Gamma$ tale che $\Gamma \vdash \bot$.

Dato che $\Sigma$ non può essere vuoto abbiamo che $\Sigma = \{\varphi_1, \dots, \varphi_n\}$.

Esistono, quindi, $n$ elementi $\Gamma_{i_1}, \dots, \Gamma_{i_n}$ non necessariamente diversi tali che $\varphi_1 \in \Gamma_{i_1}, \dots, \varphi_n \in \Gamma_{i_n}$.

Sia $m = max\{i_1, \dots, i_n\}$, ovviamente $\Sigma \subseteq \Gamma_m$ e, quindi, $\Gamma_m \vdash \bot $ contraddicendo la seconda ipotesi. \\[0.1ex]

Dimostriamo che $\Gamma$* è \textit{massimale}.

Supponiamo che esista $\Delta \neq \Gamma$* consistente tale che $\Gamma$* $\subseteq \Delta$.

Per un certo indice $k$, $\varphi_k \in \Delta - \Gamma$*, e per costruzione $\varphi_k \in \Gamma_{k+1}$ (altrimenti $\Gamma_k \cup \{\varphi_k\} \vdash \bot$) e conseguentemente $\Delta \vdash \bot$, e quindi $\varphi_k \in \Gamma$* contraddicendo l'assunzione.}

Gli insiemi massimali godono di importanti \textbf{proprietà}.

\Proposizione{\textit{Sia} $\Gamma$ \textit{massimale consistente allora}:
\begin{enumerate}
  \item $\Gamma \vdash \varphi \Leftrightarrow \varphi \in \Gamma$: insiemi massimali consistenti sono chiusi per derivabilità
  \item $\forall \varphi, \varphi \in \Gamma$ oppure $\neg \varphi \in \Gamma$
  \item $\varphi \land \psi \in \Gamma \Leftrightarrow (\varphi \in \Gamma \ \text{e} \ \psi \in \Gamma)$
  \item $\forall \varphi \ \text{e} \ \psi, \varphi \rightarrow \psi \in \Gamma \Leftrightarrow (\varphi \in \Gamma \rightarrow \psi \in \Gamma)$
\end{enumerate}}{\\[0.2ex]

\begin{enumerate}
  \item supponiamo che $\varphi \not \in \Gamma$, per la massimamilità di $\Gamma$ abbiamo che $\Gamma, \varphi \vdash \bot$, e oer le regole dell'implicazione $\Gamma \vdash \neg \varphi$ e, quindi, $\Gamma$ è inconistente $\Rightarrow$ \textbf{assurdo}.
  
  Vicerversa, per definizione della relazione $\vdash$ tenendo conto che $\varphi$ è una derivazione, se $\varphi \in \Gamma$, allora $\Gamma \vdash \varphi.$
  \item se $\varphi \in \Gamma$, allora per la massimalità di $\Gamma$ abbiamo che $\Gamma, \varphi \vdash \bot$ e, quindi, $\Gamma \vdash \neg \varphi$ e per la prima ipotesi $\neg \varphi \in \Gamma$

\textbf{\hl{Metodo diretto}}\\
Si basa sull’estensione dei risultati dimostrati nella Proposizione 9 e nel Lemma 10.

\textbf{\hl{Metodo indiretto}}\\

  \item ($\varphi \lor \psi$)* = $\lnot(\lnot \varphi* \land \lnot \psi*)$ 
\end{itemize}

\chapter*{Logica del primo ordine}
\addcontentsline{toc}{chapter}{Logica del primo ordine}

\example{}{\textit{\hl{Tutti} i predicati sono positivi, 9 è un quadrato, quindi 9 è positivo}. \\[0.2ex]
$\Rightarrow$ \textbf{tutti} = quantificatore \\[0.3ex]
Asserzioni di questo tipo, non posso essere espresse dalla logica proposizionale.}

\example{}{\textit{Tutti i numeri \hl{pari} sono \hl{somma} di numeri primi dispari.} \\[0.2ex]
$\Rightarrow$ \textbf{pari} = proprietà \\[0.2ex]
$\Rightarrow$ \textbf{somma} = relazione tra numeri \\[0.3ex]
Nella logica proposizionale non abbiamo la possibiltà di esprimeere proprietà e e relazioni.}

La \hl{logica classica proposizionale ha un basso potere espressivo}, insufficiente a formalizare le strutture matematiche. \\[0.2ex]
Ecco che la \hl{\textbf{logica del primo ordine} (o \textbf{dei predicati}) estende la sintassi permettendo di trattare i quantificatori $\exists$ e $\forall$}, rispettivamente con il significato di \textit{esiste} e \textit{per ogni}. \\[0.2ex]
I quantificatori $\forall$ e $\exists$ permettono di esprimere proprietà e relazioni che prima non potevano essere descritte.

\section*{Sintassi}
\addcontentsline{toc}{section}{Sintassi}
Il linguaggio della logica del primo ordine è basato su un alfabeto composto da diverse entità distinte.

\definizione{alfabeto della logica del primo ordine}{$\Rightarrow$ \textbf{connettivi}: $\{\land, \lor, \rightarrow, \lnot\}$ \\[0.2ex]

$\Rightarrow$ \textbf{quantificatori}: $\{\forall, \exists\}$ \\[0.2ex]

$\Rightarrow$ \textbf{simboli di variabili}: insieme numerabile $\text{Var} = \{v_0, v_1, \dots\}$ denotate con le etichette $x, y, z$ ed eventualmente indicizzate \\[0.2ex]. \\[0.2ex]
$\Rightarrow$ \textbf{simboli di relazione}: simboli n-ari (arità) coinvolti $\forall n \in \mathbb{N} \ \backslash \{0\}$ in un insieme $\mathcal{R}^{(n)}$
\begin{description}[leftmargin=2cm]
  \item [$\mathcal{R}^{(1)}$]: relazione unaria (proprietà)
  \item [$\mathcal{R}^{(2)}$]: relazione binaria
  \item [$\mathcal{R}^{(3)}$]: relazione ternaria
  \item [$\vdots$]
\end{description}

$\Rightarrow$ \textbf{simboli di funzione}: simboli n-ari (arità) coinvolti $\forall n \in \mathbb{N} \ \backslash \{0\}$ in un insieme $\mathcal{F}^{(n)}$
\begin{description}[leftmargin=2cm]
  \item [$\mathcal{F}^{(1)}$]: funzione unaria
  \item [$\mathcal{F}^{(2)}$]: funzione binaria
  \item [$\mathcal{F}^{(3)}$]: funzione ternaria
  \item [$\vdots$]
\end{description}

$\Rightarrow$ \textbf{simbolo opzionale}: relazione binario non appartenente a $\mathcal{R}^{(2)}$ denotato con $=$ (uguale).\\
Se esiste un $n \geq 1$ tale che $\mathcal{F}^{(n)} \neq \emptyset$, allora $\mathcal{L}$ deve contenere $=$ \\[0.2ex]

$\Rightarrow$ \textbf{simboli ausiliari}: parentesi tonde, virgole, punti ecc.}

L'\hl{insieme delle stringhe formate con i simboli dell'alfabeto $\mathcal{L}$} della logica del primo ordine \hl{è chiamato \textbf{$\mathcal{L}\text{-stringhe}$}}.

Le \hl{\textbf{costanti} sono gli elementi di $\mathcal{F}^{(0)}$} (funzioni di arità 0) e che vengono denotate con $\mathcal{C}$. \\[0.4ex]

A differenza del linguaggio proposizionale, dato dalle sole formule logiche, il linguaggio del primo ordine possiede dei \textbf{termini} e a partire da questi si ottiene un linguaggio delle \textbf{formule logiche}.

\definizione{$\mathcal{L}$-termini}{Dato l'alfabeto $\mathcal{L}$, l'insieme dei termini $\text{TERM}_{\mathcal{L}}$ è il più piccolo insieme X tale che:
\begin{itemize}
  \item $\text{Var} \in \text{X}$
  \item $\mathcal{C} \in \text{X}$
  \item se $t_1, \dots, t_n \in \text{X}$ e $f$ è un simbolo fi funzione n-aria, allora $f(t_1, \dots, t_n) \in \text{X}$
\end{itemize}}

\definizione{$\mathcal{L}$-formule}{Dato l'alfabeto $\mathcal{L}$, l'insieme delle formule $\text{FORM}_{\mathcal{L}}$ è il più piccolo insieme X tale che:
\begin{itemize}
  \item $\bot \in \text{X}$
  \item se $t_1, \dots, t_n \in \text{TERM}_{\mathcal{L}}$, allora $t_1 = t_2 \in \text{X}$
  \item se $P$ è una relazione k-aria e $t_1, \dots, t_k \in \text{TERM}_{\mathcal{L}}$, allora $P(t_1, \dots, t_k) \in \text{X}$
  \item se $\varphi, \psi \in \text{X}$, allora $\varphi \circ \psi \in \text{X}$, con $\circ \in \{\land, \lor, \rightarrow\}$
  \item se $x \in \text{Var}$ e $\varphi \in \text{X}$, allora $Qx.(\varphi) \in X$, con $Q \in \{\forall, \exists\}$
\end{itemize}}

Per convenzione i quantificatori $\forall, \exists$ legano più di ogni altro connettivo. \\[0.2ex]
Di conseguenza, per esempio, la formula $\forall y. \varphi \land \psi$ è da intendersi come $(\forall y. \varphi) \land \psi$. \\[0.2ex]
Se, invece, si vuole quantificare l'intera congiunzione, allora si deve scrivere $\forall y. (\varphi \land \psi)$.

\definizione{sottoformule}{E' una funzione $\text{SF}: \text{FORM}_{\mathcal{L}} \rightarrow 2^{\text{FORM}_{\mathcal{L}}}$, la quale prende in input una formula e restituisce l'insieme delle sue sottoformule. \\[0.2ex]
E' definita come segue:
\begin{itemize}
  \item $\text{SF}(\bot) = \{\bot\}$
  \item $\text{SF}[t_1 = t_2] = \{t_1 = t_2\}$
  \item $\text{SF}[P(t_1, \dots, t_k)] = \{P(t_1, \dots, t_k)\}$
  \item $\text{SF}(\varphi \circ \psi) = \text{SF}[\varphi] \cup \text{SF}[\psi] \cup \{\varphi \circ \psi\}$, con $\circ \in \{\land, \lor, \rightarrow\}$
  \item $\text{SF}[Qx.(\varphi)] = \text{SF}[\varphi] \cup \{Qx.(\varphi)\} \cup \text{SF}(\varphi)$, con $Q \in \{\forall, \exists\}$
\end{itemize}}

\subsection*{Variabili e sostituzioni}
\addcontentsline{toc}{subsection}{Variabili e sostituzioni}
All'interno di una formula le variabili possono essere \textbf{libere} e/o \textbf{legate}.

\definizione{variabile libera (induttiva)}{Sull'insieme dei termini $\text{TERM}_\mathcal{L}$:
\begin{itemize}
  \item $\text{F}[c] = \emptyset$
  \item $\text{F}[x] = \{x\}$
  \item $\text{F}[f(t1_, \dots, t_n)] = \text{FV}[t_1] \cup \text{FV}[t_2] \cup \dots \text{FV}[t_n]$
\end{itemize}

Sull'insieme delle formule $\text{FORM}_\mathcal{L}$:
\begin{itemize}
  \item $\text{FV}[\bot] = \emptyset$
  \item $\text{FV}[t_1 = t_2] = \text{FV}[t_1] \cup \text{FV}[t_2]$
  \item $\text{FV}[P(t_1, \dots, t_k)] = \text{FV}[t_1] \cup \text{FV}[t_2] \cup \dots \cup \text{FV}[t_k]$
  \item $\text{FV}[\varphi \circ \psi] = \text{FV}[\varphi] \cup \text{FV}[\psi]$
  \item $\text{FV}[Qx.(\varphi)] = \text{FV}[\varphi] \ \backslash \ \{x\}$, con $Q \in \{\forall, \exists\}$
\end{itemize}}

In modo formale, il concetto di \textbf{variabile libera} è definito come:
\begin{equation*}
  \text{FV}[\varphi] = \{x \in \text{Var} : x \text{ è libera in } \varphi\}
\end{equation*}

Un termine $t \in \text{TERM}_\mathcal{L}$ si dice \textbf{chiuso} se $\text{FV}[t] = \emptyset$. \\[0.2ex]
Invece, una formula $\varphi \in \text{FORM}_\mathcal{L}$ si dice \textbf{chiusa} (o \textbf{sentenza}) se $\text{FV}[\varphi] = \emptyset$.

\definizione{sostituzione di una variabile (ricorsiva)}{La sostituzione di una variabile $x$ con un termine $t$ in un termine $u$ (denotata con $u[x/t]$) è definita come:
\begin{itemize}
  \item $c[x/t] = c$
  \item $x[x/t] = t$
  \item $y[x/t] = \begin{cases}
    t & \text{se } y = x \\
    y & \text{se } y \neq x
  \end{cases}$
  \item $f(t_1, \dots, t_n)[x/t] = f(u_1[x/t], \dots, u_n[x/t])$
\end{itemize}}

\definizione{sostituzione di una variabile (induttiva)}{La sostituzione di una variabile $x$ con un termine $t$ in una formula $\varphi$ (denotata con $\varphi[x/t]$) è definita assumendo che $t$ è \textbf{libero per} $x$ in $\varphi$:
\begin{itemize}
  \item $\bot[x/t] = \bot$
  \item $(t_1 = t_2)[x/t] = t_1[x/t] = t_2[x/t]$
  \item $R(t_1, \dots, t_k)[x/t] = R(t_1[x/t], \dots, t_k[x/t])$
  \item $(\varphi \circ \psi)[x/t] = \varphi[x/t] \circ \psi[x/t]$, con $\circ \in \{\land, \lor, \rightarrow\}$
  \item $Q \in \{\forall, \exists\} = \begin{cases}
    Qx.(\varphi)[x/t] = (Qx.\varphi) & \text{se } y = x \\
    Qy.(\varphi)[x/t] = (Qy.\varphi[x/t]) & \text{se } y \neq x
  \end{cases}$
\end{itemize}}

Esistono due tipi di vincoli per le sostituzioni:
\begin{enumerate}
  \item sintatticamente non si può sostituire varibaili legate da quantificatori
  \item non tutti i termini possono essere utilizzati
\end{enumerate}
