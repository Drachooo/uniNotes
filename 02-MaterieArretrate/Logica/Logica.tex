  \chapter*{Introduzione alla logica}
  \addcontentsline{toc}{chapter}{Introduzione alla logica}
  La \textbf{logica} si occupa di \hl{formalizzare i ragionamenti matematici} utilizzando il linguaggio matematico.

  Ciò è dovuto al fatto che il linguaggio naturale è ambiguo per la descrizione di teorimi.

  La logica si divide in:
  \begin{itemize}
    \item \textbf{logica formale}: che a sua volta si divide in
    \begin{itemize}
      \item \textit{logica simbolica}
      \item \textit{Logica matematica}: studia la calcolabilità [studiata a fondamenti] 
    \end{itemize}
    \item \textbf{logica filosofica}: ha a che fare con il linguaggio naturale
  \end{itemize}

  \section*{Numeri naturali}
  \addcontentsline{toc}{section}{Numeri naturali}
  Dal punto di vista intuitivo i numeri sono un insieme infinito:
  \begin{equation*}
    \mathbb{N} = \{0, 1, 2, \dots\}
  \end{equation*}
  per i quali sono definite le operazioni standard aritmetiche elementari.

  Peano definì in maniera rigorosa i numeri naturali.

  \definizione{numeri naturali}{I \textbf{numeri naturali} sono una tupla:
    \begin{equation*}
      \langle \mathbb{N}, 0, succ \rangle
    \end{equation*}

    Su quetsa tupla, sono definite 4 assiomi (teoremi che non si dimostrano):
    \begin{enumerate}
      \item $0 \in \mathbb{N}$
      \item $succ: \mathbb{N} \rightarrow \mathbb{N}$ (operazione unaria iniettiva)
      \item $0 \notin \mathrm{Im}(succ)$ (zero non è successore di nessuno)
      \item (\textbf{Principio di induzione}) per ogni $\mathbb{P} \subseteq \mathbb{N}$ valgono le seguenti proprietà:
      \begin{itemize}
        \item $0 \in \mathbb{P}$
        \item $\forall n \in \mathbb{N}, (n \in \mathbb{P} \rightarrow succ(n) \in \mathbb{P})$
      \end{itemize}
      $\Rightarrow \mathbb{P} = \mathbb{N}$
    \end{enumerate}}


  \chapter*{Logica proposizionale}
  \addcontentsline{toc}{chapter}{Logica proposizionale}
  
  La logica è lo studio del ragionamento deduttivo su delle entità linguistiche che indentifichiamo essere le proposizioni.\\
  La logica classica è la branca che studia le poposizioni: frasi che possono avere due valori di verità, \textbf{VERO} o \textbf{FALSO}.

  \definizione{Alfabeto $\mathcal{L}$}{L'alfabeto $\mathcal{L}$ della logica prop ha i seguenti simboli:
  \begin{itemize}
    \item insieme numberabile di simboli proposizionali: \textbf{$AT$}$_{prop}$ = \{$p_0$, $p_1$, \dots\}
    \item connettivi: $\land$, $\lor$, $\rightarrow$, $\bot$
    \item simboli ausiliari: servono per costruire il linguaggio ma non ne fanno parte, per esempio le parentesi tonde ()
  \end{itemize}
  }
  Le proposizioni semplici vengono definite \textbf{atomi} oppure \textbf{proposizioni atomiche}.

  \definizione{Appartenenza insieme prop}{L'insieme prop delle proposizioni è definito come il più piccolo insieme P di stringhe su $\mathcal{L}$ tali che:
  \begin{itemize}
    \item $AT_{prop}$ $\subseteq$ P.
    \item $\bot \in$ P.
    \item se $\phi, \psi \in$ P allora $(\phi \land \psi), (\phi \lor \psi), (\phi \rightarrow \psi) \in P$
  \end{itemize}}

  \textbf{Esempio:}\\
  $(p_1 \rightarrow p_2)$ $\in$ PROP \qquad\qquad\qquad \textcolor{green!50!black}{$\rightarrow$ Completa}\\
  $p_7 \lor p_4 \notin$ PROP \qquad\qquad\qquad\qquad \textcolor{red}{$\rightarrow$ Mancano le parentesi}\\
  $((\bot \lor p_7) \land (\lnot p_4)) \in$ PROP \qquad\qquad \textcolor{green!50!black}{$\rightarrow$ Completa}\\

  \notazione{Chiamiamo AT = $AT_{prop} \cup \{\bot\}$
  \begin{enumerate}[label= \alph*)]
    \item $\lnot \phi$ denotiamo $(\phi \rightarrow \bot)$
    \item $\phi \equiv \psi$ denotiamo $((\phi \rightarrow \psi) \land (\psi \rightarrow \phi))$
  \end{enumerate}
  }

  \convenzione{
  \begin{enumerate}
    \item Omettiamo le parentesi più esterne \qquad $(\psi \rightarrow \phi)$ diventa $\psi \rightarrow \phi$
    \item Scriviamo $\lnot \psi$ anzichè $(\lnot \psi)$
    \item $\land$ e $\lor$ legano più forte di $\rightarrow$ e $\equiv$ \quad $\phi_1 \land \phi_2 \rightarrow \psi_1 \lor \psi_2$ sta per $(\phi_1 \land \phi_2) \rightarrow (\psi_1 \lor \psi_2)$
  \end{enumerate}
  }


  \subsection*{Funzioni ricorsive su Prop}
  \addcontentsline{toc}{subsection}{Funzioni ricorsive}
  \definizione{Funzione $\mathscr{L}$ lunghezza}{Funzione $\mathscr{L}$ :PROP $\rightarrow \mathbb{N}$ che ad ogni $\phi \in PROP$ associa la sua lunghezza
  \vspace{0,2cm}
  \begin{enumerate}
    \item $\mathscr{L}$($\phi$) = 1 \qquad se $\phi \in AT$
    \item $\mathscr{L}$($\phi$ o $\psi$) = 3 + $\mathscr{L}(\phi)$ + $\mathscr{L}(\psi)$ \qquad con o $\in$ $\{\land, \lor, \rightarrow\}$
    \item $\mathscr{L}$($\lnot \phi$) = 3 + $\mathscr{L}(\phi)$
  \end{enumerate}
  il \textbf{+3} indica le due parentesi tonde e il connettivo
  }  

  \esercizio[Definiamo il numero di parentesi tonde in una formula]{b($\varphi$) $\rightarrow$ numero di parentesi in una formula proposizionale
  \begin{itemize}
    \item \textcolor{red}{b(}$\varphi$\textcolor{red}{)} = 0 \qquad  \qquad \qquad con $\varphi \in$ AT
    \item \textcolor{red}{b(}($\varphi$ o $\phi$)\textcolor{red}{)} = 2 + b($\varphi$) \qquad con o $\in$ \{$\land, \lor, \rightarrow$ \}
    \item \textcolor{red}{b(}($\lnot \varphi$)\textcolor{red}{)} = 2 + b($\varphi$) 
  \end{itemize}}

  \esercizio[Definizione di parsing tree]{Definiamo il parsing tree di una proposizione
  \begin{itemize}
    \item T($\varphi$) =  $\cdot \varphi$     \qquad con $\varphi$ $\in$ AT
    \item T($\varphi$ o $\psi$) = $\cdot$(T($\varphi$), T($\psi$))    \qquad \qquad con o $\in$ \{ $\land$, $\lor$, $\rightarrow$\}
    \item T($\lnot \varphi$) = $\lnot$(T($\varphi$))
  \end{itemize}
  }
  

  Una volta definito costruiamo il \textit{parsing tree}\footnote{Parsing tree: visualizzazione ad albero di una stringa} della proposizione\\
  \begin{forest}
    [$(p_1 \rightarrow (\bot \lor (\lnot p_3)))$
      [$p_1$]
      [$\bot \lor (\lnot p_3)$
        [$\bot$]
        [$\lnot p_3$
        [$p_3$]
        ]
      ]
    ]
  \end{forest}


  \definizione{Rango $r$ di una formula}{
    \begin{itemize}
      \item $r$($\varphi$) = 0  \qquad  \qquad $\varphi \in$ AT
      \item $r$($\varphi$ o $\psi$) = 1 + max\{$r$($\varphi$), $r$($\psi$)\} \qquad con o $\in$ \{$\lor$, $\land$, $\rightarrow$\}
      \item $r$($\lnot \varphi$) = 1 + r($\varphi$)
    \end{itemize}
  }

  \definizione{Sottoformula Sub}{
    \begin{itemize}
      \item Sub($\varphi$) = \{$\varphi$\}   \qquad $\varphi \in$ AT
      \item Sub($\varphi$ o $\psi$) = Sub($\varphi$) $\cup$ Sub($\psi$) $\cup$ \{$\varphi$ o $\psi$\} \qquad con o $\in$ $\{\land, \lor, \rightarrow\}$
      \item Sub($\lnot \varphi$) = Sub($\varphi$) $\cup$ \{($\lnot \varphi$)\} 
    \end{itemize}
  }

  $r$($\varphi \land \psi$) = 1 + max\{2,1\} = 1+2 = 3}

      \item \#($\varphi \lor \psi$) = 1 + \#$\varphi$ + \#$\psi$
      \item \#($\lnot \varphi)$ = 1 + \#$\varphi$
    \end{itemize}
  }

  \section*{Semantica}
  \addcontentsline{toc}{section}{Semantica}
  La \textbf{semantica} si occupa di \hl{attribuire un significato sintattico} alle proposizioni (vero - 1 o falso - 0).


  \definizione{valutazione di verità}{E' una funzione $\sigma: PROP \rightarrow \{{0, 1}\}$ tale che:
    \begin{itemize}
      \item $\sigma(\bot) = 0$
      \item $\sigma(\lnot \varphi) = 1$ se e solo se $\sigma(\varphi) = 0$
      \item $\sigma(\phi \land \psi) \Leftrightarrow \sigma(\phi) = \sigma(\psi) = 1$
      \item $\sigma(\phi \lor \psi) \Leftrightarrow \sigma(\phi) = 1$ oppure $\sigma(\psi) = 1$
      \item $\sigma(\phi \rightarrow \psi) \Leftrightarrow \sigma(\phi) = 1$ allora $\sigma(\psi) = 1$
    \end{itemize}
    }
  

  Questa definizione per la funzione di valutazione è pressochè la formulzazione matematica delle \textbf{tavole di verità}.

  \definizione{valutazione atomica}{Funzione $v: AT \rightarrow \{0, 1\}$ tali che $v(\bot) = 0$ 
  }

  \proposizione{Per ogni valutazione $v$ esiste ed è unica una \textit{valutazione di verità} ${\llbracket - \rrbracket}_v$ tale che per ogni $p \in \mathbf{AT}$, ${\llbracket p \rrbracket}_v = v(p)$.
  }

  \definizione{concordanza}{  Dato un insieme finito $P$ di simboli proposizionali, si dice che due valutazioni $v, v'$ \textbf{concordano} su $P$ se $v_{|P} = v'_{|P}$. 
  }

  La proposizione precedente ci dice che l'insieme di tutte le possibili valutazione ha le stessa cardinalità di $\mathbb{N}$.

  Un'altra proprietà delle valutazione è stabilita dalla seguente proposizione.

  \proposizione{Se $P(\phi)$ è l'insieme dei simboli proposizionali che occorrono in una formula $\phi$ e $v, v'$ due valutazioni che concordano su $P(\phi)$ allora ${\llbracket \phi \rrbracket}_v = {\llbracket \phi \rrbracket}_{v'}$.
  }

  Se un problema è vero in tutte le sue valutazioni, allora è \textbf{decidibile} e viene detto \textbf{tautologia}.

  Si estende la valutazione di verità a un insieme arbitrario di proposizioni $\Gamma$: \par
  \begin{center}
      ${\llbracket \Gamma \rrbracket}_v = 1$ se e solo se per ogni $\alpha \in \Gamma$, ${\llbracket \alpha \rrbracket}_v = 1$
  \end{center}

  Tuttavia si pone attenzione al fatto che se ${\llbracket \phi \rrbracket}_v \neq 1$, allora non significa che per ogni $\alpha \in \Gamma$ si ha ${\llbracket \alpha \rrbracket}_v = 0$, al contrario si intende che esiste almeno un $\alpha \in \Gamma$ tale per cui ${\llbracket \alpha \rrbracket}_v = 0$.

  Viceversa è più forte affermare che ${\llbracket \Gamma \rrbracket}_v = 0$, in quanto per questo caso ogni $\alpha \in \Gamma$ ha valutazione ${\llbracket \alpha \rrbracket}_v = 0$.

  Partendo dal concetto di valutazione dell formule, è possibile definire cosa significhi che da un insieme $\Gamma$ di proposizioni \textbf{segue logicamente} una proposizione $\varphi$, o in altre parole che $\varphi$ sia \textbf{conseguenza logica} di $\Gamma$.

  \definizione{conseguenza logica}{Dato un insieme $\Gamma$ di formule, si dice che la formual $\varphi$ è conseguenza logica dalle formule in $\Gamma$ e si scrive $\Gamma \models \varphi$, se per ogni valutazione $v$, ${\llbracket \Gamma \rrbracket}_v = 1 \Rightarrow {\llbracket \varphi \rrbracket}_v = 1$.
  }

  La formula $\varphi$ è detta \textbf{soddisfacibile} se esiste una valutazione $v$ tale che ${\llbracket \varphi \rrbracket}_v = 1$, altrimenti risulta essere \textbf{insoddisfacibile}.

  A sua volta un insieme di formule $\Gamma$ è soddisfacibile se esiste una valutazione $v$ tale che per ogni $\varphi \in \Gamma$ si ha ${\llbracket \Gamma \rrbracket}_v = 1$, altrimenti si dice insoddisfacibile.

  \notazione{\begin{itemize}
        \item si scrive $\models \varphi$ al posto di $\phi \models \varphi$
        \item si scrive $\psi_1, \dots, \psi_n \models \varphi$ al posto di $\{\psi_1, \dots, \psi_n\} \models \varphi$
        \item $\Gamma, \Sigma, \Delta$ sono insiemi arbitrari di formule
        \item si scrive $\Gamma, \Sigma, \psi \models \varphi$ al posto di $\Gamma \cup \Delta \cup \{\psi\} \models \varphi$ 
    \end{itemize}
  }

  \subsection*{Proprietà di $\models$}
\addcontentsline{toc}{subsection}{\texorpdfstring{Proprietà di $\models$}{Proprietà di ⊨}}

  \begin{enumerate}
    \item $\Gamma \models \psi$ e $\bigtriangleup.\psi \models \varphi \Rightarrow  \Gamma, \bigtriangleup \models \varphi$
    \item $\Gamma \models \varphi \lor \psi$ e $\bigtriangleup, \varphi \models \delta$ e $\varepsilon, \psi \models \delta \Rightarrow \Gamma, \bigtriangleup, \varepsilon \models \Gamma$ 
  \end{enumerate}

  \textbf{\large Dimostrazione 1}
  \begin{itemize}[label=-]
    \item $\Gamma \models \psi \leftrightarrow$ Per ogni valutazione v.$\llbracket \Gamma \rrbracket_v = 1 \rightarrow \llbracket \psi \rrbracket_v = 1$
    \item $\bigtriangleup, \psi \models$ $\leftrightarrow$ Per ogni valutazione v.$\llbracket \bigtriangleup \rrbracket_v$ = 1 e $\llbracket \psi \rrbracket_v$ = 1 $\rightarrow \llbracket \varphi \rrbracket_v$ = 1\\
    $\Rightarrow \llbracket \Gamma, \bigtriangleup \rrbracket_v$ = 1 $\Rightarrow \llbracket \varphi \rrbracket_v$ = 1 
  \end{itemize}

  \textbf{\large Dimostrazione 2}\par
  \textcolor{orange}{Ipotesi:} \sottolinea{orange}{$\llbracket \Gamma \rrbracket_v = 1 = \llbracket \bigtriangleup \rrbracket_v = \llbracket \varepsilon \rrbracket_v \Rightarrow \llbracket \delta \rrbracket_v = 1$}\par
  \hspace*{1,40cm} $\llbracket \varphi \lor \psi \rrbracket_v = 1$\par
  \hspace*{1,40cm} $\llbracket \varphi \rrbracket_v = 1$ OR $\llbracket \psi \rrbracket_v = 1$\par

  Bisogna procedere per casi:
  \begin{itemize}
    \item Se $\llbracket \varphi \rrbracket_v = 1$ allora siccome $\llbracket \bigtriangleup \rrbracket_v = 1$ allora $\llbracket \delta \rrbracket_v = 1$
    \item Se $\llbracket \psi \rrbracket_v = 1$ allora siccome $\llbracket \varepsilon \rrbracket_v = 1$ allora $\llbracket \delta  \rrbracket_v = 1$
  \end{itemize}

  \vspace{0,5cm}

  \sottolinea{gray}{$\varphi_1 \land \varphi_2 \leftrightarrow \varphi_1' \land \varphi_2'$} che è come scrivere \sottolinea{gray}{$\varphi_1 \land \varphi_1' \leftrightarrow \varphi_2 \land \varphi_2'$}]{
    \begin{itemize}

  \definizione{Sostituzione}{Siano $\psi$ e $\gamma$ due formule e sia p $\in$ Prop. Si definisce $\psi[\textbf{p}/\gamma]$ la formula ottenuta sostituendo nella stringa $\psi$ ogni occorrenza di \textbf{p} con la stringa corrispondente a $\psi$.}

  \teorema[Sostituzione]{Siano $\varphi, \gamma_1. \gamma_2 \in$ Prop e sia p una proposizione atomica.\\
  Se $\models \gamma_1 \leftrightarrow \gamma_2$ allora $\models \varphi[p/\gamma_1] \leftrightarrow \varphi[p/\gamma_2]$}{(per induzione) su $\varphi$\\
  \sottolinea{black}{Base: Sia $\varphi \in \textbf{AT}$. Abbiamo due casi:}
  \begin{enumerate}
    \item se $\varphi \neq \textbf{p} $ allora $\varphi [\textbf{p}/\gamma_1] = \varphi[\textbf{p}/\gamma_2] = \varphi$ e quindi ovviamente $\models \varphi \equiv \varphi$
    \item se $\varphi = \textbf{p}$ allora $\varphi[\textbf{p}/\gamma_1] = \gamma_1$ e $\varphi[\textbf{p}/\gamma_2] = \gamma_2$ e quindi per ipotesi $\models \varphi[\textbf{p}/\gamma_1] \equiv \varphi[\textbf{p}/\gamma_1]$
  \end{enumerate}
  
  \vspace{2em}
  
  \sottolinea{black}{Passo induttivo: Esaminiamo i vari casi}
  \begin{description}
    \item[($\varphi = \varphi_1 \rightarrow \varphi_2$)] Per ipotesi induttiva:\\ $\models \varphi_1[\textbf{p}/\gamma_1] \equiv \varphi_1[\textbf{p}/\gamma_2]$ e $\models \varphi_2[\textbf{p}/\gamma_1] \equiv \varphi_2[\textbf{p}/\gamma_2].$\\
    Ora abbiamo che $\models \varphi_1[\textbf{p}/\gamma_1] \rightarrow \varphi[\textbf{p}/\gamma_1] \equiv \varphi_1[\textbf{p}/\gamma_2] \rightarrow \varphi_2[\textbf{p}/\gamma_2].$\\ 
    Dato che $\varphi[\textbf{p}/\gamma_1] = \varphi_1[\textbf{p}/\gamma_1] \rightarrow \varphi_2[\textbf{p}/\gamma_1]$ e $\varphi[\textbf{p}/\gamma_2] = \varphi_1[\textbf{p}/\gamma_2] \rightarrow \varphi_2[\textbf{p}/\gamma_2]$
    \item[($\varphi = \varphi_1 \land \varphi_2$), ($\varphi = \varphi_1 \lor \varphi_2$)] Il ragionamento è analogo al caso precedente
  \end{description}
  }

    \item Metodo 1: Utilizzare le tavole di verità $\rightarrow$ dispendioso
    \item Metodo 2: Utilizzare la semantica $\rightarrow$ da utilizzare in esame
  \end{itemize}
